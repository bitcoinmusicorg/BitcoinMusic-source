file(GLOB HEADERS "include/muse/manifest/*.hpp")

################# external plugins ###################

# external_plugins target depends on all plugins
add_library( muse_external_plugins
   external_plugins.cpp
   )

# generate file based on comments
file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_external_plugins.inc" "// this file is autogenerated\n#pragma once\n#define MUSE_EXTERNAL_PLUGIN_LIST \\\n   ")

foreach(pin $ENV{MUSE_EXTERNAL_PLUGINS})
   file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_external_plugins.inc" "( ${pin} )")
   target_link_libraries( muse_external_plugins "${pin}" )
endforeach()

file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_external_plugins.inc" "\n")

################# internal plugins ###################

add_library( muse_internal_plugins
   internal_plugins.cpp
   )

file(WRITE "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_internal_plugins.inc" "// this file is autogenerated\n#pragma once\n#define MUSE_INTERNAL_PLUGIN_LIST \\\n   ")

foreach(pin $ENV{MUSE_INTERNAL_PLUGINS})
   file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_internal_plugins.inc" "( ${pin} )")
   target_link_libraries( muse_internal_plugins "muse_${pin}" )
endforeach()

file(APPEND "${CMAKE_CURRENT_BINARY_DIR}/inc/mf_internal_plugins.inc" "\n")

################# common ###################

add_library( muse_mf_plugins
   mf_plugins.cpp
   ${HEADERS}
   ${CMAKE_CURRENT_BINARY_DIR}/inc/mf_internal_plugins.inc
   ${CMAKE_CURRENT_BINARY_DIR}/inc/mf_external_plugins.inc
   )
target_link_libraries( muse_mf_plugins fc )

target_include_directories( muse_mf_plugins
   PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include"
   PRIVATE "${CMAKE_CURRENT_BINARY_DIR}/inc" )

INSTALL( TARGETS
   muse_mf_plugins

   RUNTIME DESTINATION bin
   LIBRARY DESTINATION lib
   ARCHIVE DESTINATION lib
)
